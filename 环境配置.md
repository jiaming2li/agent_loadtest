#### 3. create fake nodes
```
fake-node.yaml(先建 1 个验证,再 scale):

apiVersion: v1
kind: Node
metadata:
  annotations:
    kwok.x-k8s.io/node: fake          # ← kwok 接管这个节点
    node.alpha.kubernetes.io/ttl: "0"
  labels:
    type: kwok                        # ← 给 nodeSelector 用
  name: kwok-node-0
spec:
  taints:
    - key: kwok.x-k8s.io/node
      value: fake
      effect: NoSchedule              # ← 假节点默认 taint
status:
  allocatable: {cpu: "1000", memory: 1Ti, pods: "1000000"}
  capacity:    {cpu: "1000", memory: 1Ti, pods: "1000000"}
  conditions:
    - {type: Ready, status: "True", reason: KubeletReady}
  nodeInfo: {kubeletVersion: fake}
```

#### 5. create sandboxset

```
apiVersion: agents.kruise.io/v1alpha1
kind: SandboxSet          # ← 池子,不是单个 sandbox
metadata: {name: loadtest, namespace: default}
spec:
  replicas: 3             # ← 声明"要 3 个待领沙箱"
  template:
    spec:
      nodeSelector: {type: kwok}      # 钉到假节点
      tolerations: [{key: kwok.x-k8s.io/node, operator: Exists, effect: NoSchedule}]
      containers:
        - {name: sandbox, image: busybox:latest, resources: {requests: {cpu: 100m, memory: 64Mi}}}

kubectl apply -f /tmp/loadtest-sandboxset.yaml
```
